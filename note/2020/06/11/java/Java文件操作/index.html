private Path fileHomePath;

public void before() {
    fileHomePath = Paths.get("d:/test");

创建目录

@Test
public void createDir() throws IOException {
    Path path = Files.createDirectories(fileHomePath.resolve(String.format("dir%s", System.currentTimeMillis())));
    System.out.println(String.format("目录%s创建成功", path));
}

创建文件

@Test
public void createFileOnPath() throws IOException {
    Path dirPath = Files.createDirectories(fileHomePath.resolve(String.format("dir%s", System.currentTimeMillis())));
    Path filePath = Files.createFile(dirPath.resolve("test.txt"));
    System.out.println(String.format("文件%s创建成功", filePath));
}

创建文件并写入内容

@Test
public void createFile() throws IOException {
    Path filePath = fileHomePath.resolve("hello.txt");

    if (!Files.exists(filePath)) {
        // 文件不存在
        Files.createFile(filePath);
    }

    // 向文件写入内容
    File file = new File(filePath.toUri());
    try (FileOutputStream fileOutputStream = new FileOutputStream(file);) {
        fileOutputStream.write("hello...".getBytes());
    }

}

向文件追加内容

@Test
public void writeFile() throws IOException {
    Path filePath = fileHomePath.resolve("hello.txt");
    Files.write(
        filePath,
        String.format("now:%s#", System.currentTimeMillis()).getBytes(),
        StandardOpenOption.APPEND, StandardOpenOption.CREATE);
}

删除文件

@Test
public void deleteFile() throws IOException {
    Path filePath = fileHomePath.resolve("hello.txt");
    Files.deleteIfExists(filePath);
}

遍历文件夹

@Test
public void list() throws IOException {

    try (
        Stream<Path> pathStream = Files.list(fileHomePath)
    ) {
        pathStream.forEach(System.out::println);
    }

}

递归遍历文件夹

  • 递归遍历文件夹，找到匹配内容的文件
@Test
public void findMatchFile() throws IOException {
    final String key = "hello";
    try (
        Stream<Path> pathStream = Files.find(
            fileHomePath,
            10,
            (path, fileAttributes) -> {
                if (Files.isDirectory(path)) return false;
                try {
                    String txt = new String(Files.readAllBytes(path));
                    if (StringUtils.contains(txt, key)) {
                        return true;
                    }
                } catch (IOException e) {
                    e.printStackTrace();
                }
                return false;
            }

        )
    ) {
        pathStream.forEach(System.out::println);
    }

}
