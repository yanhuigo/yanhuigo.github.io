Zookeeper是什么

  • 从本质上可以理解成一个可以感知节点变化的文件系统(节点比较类似Linux文件系统)
  • 用于协调分布式组件之间的活动，基于节点感知的特性可解决分布式环境中服务发现，配置一致性…相关问题
  • 这篇文章介绍的很详细

环境搭建

  1. 获取安装包zookeeper-3.4.6.tar.gz
  2. 解压 tar -zxvf zookeeper-3.4.6.tar.gz
  3. 创建配置文件 cp conf/zoo_sample.cfg conf/zoo.cfg
  4. 启动服务 bin/zkServer.sh start
  5. 启动客户端连接 bin/zkCli.sh

Java连接zookeeper

引入依赖

<dependency>
    <groupId>org.apache.zookeeper</groupId>
    <artifactId>zookeeper</artifactId>
    <version>3.4.6</version>
</dependency>

测试代码

package cn.yanhui;

import lombok.SneakyThrows;
import lombok.extern.java.Log;
import org.apache.zookeeper.WatchedEvent;
import org.apache.zookeeper.Watcher;
import org.apache.zookeeper.ZooKeeper;
import org.apache.zookeeper.data.Stat;

import java.util.concurrent.CountDownLatch;

/**
 * @author yanhui@git.com.cn
 * @since 2020-06-22 14:44
 */
@Log
public class ZookeeperTest implements Watcher {

    private final static CountDownLatch connectedSemaphore = new CountDownLatch(1);

    private final static Stat stat = new Stat();

    public static ZooKeeper zk;

    public static void main(String[] args) throws Exception {
        // 节点路径
        final String path = "/test";

        // 连接zookeeper并指定监听器
        zk = new ZooKeeper("192.168.252.128:2181", 5000, new ZookeeperTest());

        System.out.println("等待连接成功...");
        connectedSemaphore.await();

        System.out.println("获取节点信息");
        System.out.println(new String(zk.getData(path, true, stat)));

        // 等待
        Thread.sleep(Integer.MAX_VALUE);
    }

    @SneakyThrows
    @Override
    public void process(WatchedEvent event) {
        System.out.println(event.toString());
        if (Event.KeeperState.SyncConnected == event.getState()) {  //zk连接成功通知事件
            if (Event.EventType.None == event.getType() && null == event.getPath()) {
                System.out.println("连接成功...取消主线程等待");
                connectedSemaphore.countDown();
            } else if (event.getType() == Event.EventType.NodeDataChanged) {  //zk目录节点数据变化通知事件
                System.out.println("配置已修改，新值为：" + new String(zk.getData(event.getPath(), true, stat)));
            }
        }
    }
}
